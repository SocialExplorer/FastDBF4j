mvn install:install-file -DgroupId=com.socialexplorer -DartifactId=fastdbf4j -Dversion=1.0.0 -Dfile=FastDBF4j.jar -Dpackaging=jar -DgeneratePom=true -DlocalRepositoryPath=./maven-repo -DcreateChecksum=true

=======
To add the jar as dependency in the project using maven, add this to your pom.xml (inside the project tag):

<!-- FastDBF4j dependency -->
<dependencies>
	<dependency>
		<groupId>com.socialexplorer</groupId>
		<artifactId>fastdbf4j</artifactId>
		<version>1.0.0</version>
	</dependency>
</dependencies>

<!-- SocialExplorer maven repository -->
<repositories>
	<repository>
		<id>socialexplorer</id>
		<url>https://raw.github.com/marnautovic/FastDBF4j/master/maven-repo/</url>
	</repository>
</repositories>
